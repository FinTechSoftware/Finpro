<Meta 
    title="Components/Divider"
    component="fp-divider" 
    argTypes={{
        variant: {
            options: ['primary', 'secondary', 'success', 'danger'],
            default: 'primary',
            control: { type: 'select' }
        },
        type: {
            options: ['dashed', 'dotted', 'solid', 'rounded'],

export const DividerType = (args) => html`<fp-divider
    variant=${ifDefined(args.variant)}
    type=${ifDefined(args.type)}
    size=${ifDefined(args.size)}

export const Template = (args) => html`
    <fp-divider></fp-divider>`;

export const DividerSize = (args) => html`
    ${DividerType({ size: '4xs', ...args })}
    ${DividerType({ size: '3xs', ...args })}
    ${DividerType({ size: '2xs', ...args })}
    ${DividerType({ size: 'xs', ...args })}
    ${DividerType({ size: 's', ...args })}`;

export const DividerVariants = (args) => html`
    ${DividerType({ variant: 'primary', ...args })}

export const DividerTypes = (args) => html`
    ${DividerType({ type: 'solid', ...args })}
    ${DividerType({ type: 'dashed', ...args })}
    ${DividerType({ type: 'dotted', ...args })}
    ${DividerType({ type: 'rounded', ...args })}`;

export const DividerExample = (args) => html`

## Divider Variants

Primary divider are used for main actions of the screens. It can has 4 different "kind"s. `primary`, `secondary`, `success` and `danger`.

## Divider Type

Four different types are defined for displays. These are respectively; `solid`, `dotted`, `dashed` and `rounded`.
